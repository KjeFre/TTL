00054 static Edge* getLeadingEdgeInTriangle(Edge* e) {
00055   
00056   Edge* edge = e;
00057   
00058   // Code: 3EF (assumes triangle)
00059   if (!edge->isLeadingEdge()) {
00060     edge = edge->getNextEdgeInFace();
00061     if (!edge->isLeadingEdge())
00062       edge = edge->getNextEdgeInFace();
00063   }
00064   
00065   if (!edge->isLeadingEdge()) {
00066     return NULL;
00067   }
00068   
00069   return edge;
00070 }

00075                       vector<Node*>::iterator last,
00076                       double& xmin, double& ymin,
00077                       double& xmax, double& ymax) {

00093 Edge* Triangulation::initTwoEnclosingTriangles(vector<Node*>::iterator first,
00094                                                vector<Node*>::iterator last) {
00095    
00096   double xmin, ymin, xmax, ymax;
00097   getLimits(first, last, xmin, ymin, xmax, ymax);
00098   
00099   // Add 10% of range:  
00100   double fac = 10.0;
00101   double dx  = (xmax-xmin)/fac;
00102   double dy  = (ymax-ymin)/fac;
00103   
00104   double zval = 0;
00105   Node* n1 = new Node(xmin-dx,ymin-dy,zval);
00106   Node* n2 = new Node(xmax+dx,ymin-dy,zval);
00107   Node* n3 = new Node(xmax+dx,ymax+dy,zval);
00108   Node* n4 = new Node(xmin-dx,ymax+dy,zval);
00109   
00110   // diagonal
00111   Edge* e1d = new Edge; // lower
00112   Edge* e2d = new Edge; // upper, the twin edge
00113   
00114   // lower triangle
00115   Edge* e11 = new Edge;
00116   Edge* e12 = new Edge;
00117   
00118   // upper triangle
00119   Edge* e21 = new Edge; // upper upper
00120   Edge* e22 = new Edge;
00121   
00122   // lower triangle
00123   e1d->setSourceNode(n3);
00124   e1d->setNextEdgeInFace(e11);
00125   e1d->setTwinEdge(e2d);
00126   e1d->setAsLeadingEdge();
00127   addLeadingEdge(e1d);
00128   
00129   e11->setSourceNode(n1);
00130   e11->setNextEdgeInFace(e12);
00131   
00132   e12->setSourceNode(n2);
00133   e12->setNextEdgeInFace(e1d);
00134   
00135   // upper triangle
00136   e2d->setSourceNode(n1);
00137   e2d->setNextEdgeInFace(e21);
00138   e2d->setTwinEdge(e1d);
00139   e2d->setAsLeadingEdge();
00140   addLeadingEdge(e2d);
00141   
00142   e21->setSourceNode(n3);
00143   e21->setNextEdgeInFace(e22);

00238   
00239   // Reverse operation of splitTriangle
00240   
00241   Edge* e1 = edge.getNextEdgeInFace();
00242   Edge* le = getLeadingEdgeInTriangle(e1);
00243 #ifdef DEBUG_HE
00244   if (le == NULL)
00245     errorAndExit("Triangulation::removeTriangle: could not find leading edge");
00246 #endif
00247   removeLeadingEdgeFromList(le);
00248   
00249   Edge* e2 = e1->getNextEdgeInFace()->getTwinEdge()->getNextEdgeInFace();
00250   le = getLeadingEdgeInTriangle(e2);
00251 #ifdef DEBUG_HE
00252   if (le == NULL)
00253     errorAndExit("Triangulation::removeTriangle: could not find leading edge");
00254 #endif
00255   removeLeadingEdgeFromList(le);
00256     
00257   Edge* e3= edge.getTwinEdge()->getNextEdgeInFace()->getNextEdgeInFace();
00258   le = getLeadingEdgeInTriangle(e3);
00259 #ifdef DEBUG_HE
00260   if (le == NULL)
00261     errorAndExit("Triangulation::removeTriangle: could not find leading edge");
00262 #endif
00263   removeLeadingEdgeFromList(le);
00264   
00265   // The three triangles at the node have now been removed
00266   // from the triangulation, but the arcs have not been deleted.
00267   // Next delete the 6 half edges radiating from the node
00268   // The node is maintained by handle and need not be deleted explicitly
00269   
00270   Edge* estar = &edge;
00271   Edge* enext = estar->getTwinEdge()->getNextEdgeInFace();
00272   delete estar->getTwinEdge();
00273   delete estar;  
00274   
00275   estar = enext;
00276   enext = estar->getTwinEdge()->getNextEdgeInFace();
00277   delete estar->getTwinEdge();
00278   delete estar;
00279   
00280   delete enext->getTwinEdge();
00281   delete enext;
00282   
00283   // Create the new triangle
00284   e1->setNextEdgeInFace(e2);
00285   e2->setNextEdgeInFace(e3);
00286   e3->setNextEdgeInFace(e1);

00319 bool Triangulation::removeLeadingEdgeFromList(Edge* leadingEdge) {
00320   
00321   // Remove the edge from the list of leading edges,
00322   // but don't delete it.
00323   // Also set flag for leading edge to false.
00324   // Must search from start of list. Since edges are added to the
00325   // start of the list during triangulation, this operation will
00326   // normally be fast (when used in the triangulation algorithm)
00327   list<Edge*>::iterator it;
00328   for (it = leadingEdges_.begin(); it != leadingEdges_.end(); ++it) {
00329     
00330     Edge* edge = *it;
00331     if (edge == leadingEdge) {
00332       
00333       edge->setAsLeadingEdge(false);
00334       leadingEdges_.erase(it);
00335       
00336       break;
00337     }
00338   }
00339   
00340   if (it == leadingEdges_.end())
00341     return false;
00342   
00343   return true;
00344 }

00408 list<Edge*>* Triangulation::getEdges(bool skip_boundary_edges) const {
00409   
00410   // collect all arcs (one half edge for each arc)
00411   // (boundary edges are also collected).
00412   
00413   list<Edge*>::const_iterator it;
00414   list<Edge*>* elist = new list<Edge*>;
00415   for (it = leadingEdges_.begin(); it != leadingEdges_.end(); ++it) {
00416     Edge* edge = *it;
00417     for (int i = 0; i < 3; ++i) {
00418       Edge* twinedge = edge->getTwinEdge();
00419       // only one of the half-edges
00420       
00421       if ( (twinedge == NULL && !skip_boundary_edges) ||
00422            (twinedge != NULL && ((size_t)edge > (size_t)twinedge)) )
00423         elist->push_front(edge);
00424       
00425       edge = edge->getNextEdgeInFace();
00426     }
00427   }
00428   return elist;

00433 Edge* Triangulation::splitTriangle(Edge& edge, Node& point) {
00434   
00435   // Add a node by just splitting a triangle into three triangles
00436   // Assumes the half edge is located in the triangle
00437   // Returns a half edge with source node as the new node
00438   
00439   double x = point.x();
00440   double y = point.y();
00441   double z = point.z();
00442   
00443   // e#_n are new edges
00444   // e# are existing edges
00445   // e#_n and e##_n are new twin edges
00446   // e##_n are edges incident to the new node
00447   
00448   // Add the node to the structure
00449   //Node* new_node = new Node(x,y,z);
00450 
00451   Node* new_node = &point;
00452   Node* n1 = edge.getSourceNode();
00453   Edge* e1 = &edge;
00454   
00455   Edge* e2 = edge.getNextEdgeInFace();
00456   Node* n2 = e2->getSourceNode();
00457   
00458   Edge* e3 = e2->getNextEdgeInFace();
00459   Node* n3 = e3->getSourceNode();
00460   
00461   Edge* e1_n  = new Edge;
00462   Edge* e11_n = new Edge;
00463   Edge* e2_n  = new Edge;
00464   Edge* e22_n = new Edge;
00465   Edge* e3_n  = new Edge;
00466   Edge* e33_n = new Edge;
00467   
00468   e1_n->setSourceNode(n1);
00469   e11_n->setSourceNode(new_node);
00470   e2_n->setSourceNode(n2);
00471   e22_n->setSourceNode(new_node);

00473   e33_n->setSourceNode(new_node);
00474   
00475   e1_n->setTwinEdge(e11_n);
00476   e11_n->setTwinEdge(e1_n);
00477   e2_n->setTwinEdge(e22_n);

00526 void Triangulation::swapEdge(Edge& diagonal) {
00527   
00528   // Note that diagonal is both input and output and it is always
00529   // kept in counterclockwise direction (this is not required by all 
00530   // finctions in ttl:: now)
00531   
00532   // Swap by rotating counterclockwise
00533   // Use the same objects - no deletion or new objects
00534   Edge* eL   = &diagonal;
00535   Edge* eR   = eL->getTwinEdge();
00536   Edge* eL_1 = eL->getNextEdgeInFace();
00537   Edge* eL_2 = eL_1->getNextEdgeInFace();
00538   Edge* eR_1 = eR->getNextEdgeInFace();
00539   Edge* eR_2 = eR_1->getNextEdgeInFace();
00540   
00541   // avoid node to be dereferenced to zero and deleted
00542   Handle<Node> nR = eR_2->getSourceNode();
00543   Handle<Node> nL = eL_2->getSourceNode();
00544   
00545   eL->setSourceNode(nR.getPtr());
00546   eR->setSourceNode(nL.getPtr());
00547   
00548   // and now 6 1-sewings
00549   eL->setNextEdgeInFace(eL_2);
00550   eL_2->setNextEdgeInFace(eR_1);
00551   eR_1->setNextEdgeInFace(eL);
00552   
00553   eR->setNextEdgeInFace(eR_2);
00554   eR_2->setNextEdgeInFace(eL_1);
00555   eL_1->setNextEdgeInFace(eR);
00556   
00557   Edge* leL;
00558   if (eL->isLeadingEdge())
00559     leL = eL;
00560   else if (eL_1->isLeadingEdge())
00561     leL = eL_1;
00562   else if (eL_2->isLeadingEdge())
00563     leL = eL_2;
00564   
00565   Edge* leR;
00566   if (eR->isLeadingEdge())
00567     leR = eR;
00568   else if (eR_1->isLeadingEdge())
00569     leR = eR_1;
00570   else if (eR_2->isLeadingEdge())
00571     leR = eR_2;
00572   
00573   removeLeadingEdgeFromList(leL);
00574   removeLeadingEdgeFromList(leR);
00575   addLeadingEdge(eL);
00576   addLeadingEdge(eR);
00577 }

00631 //--------------------------------------------------------------------------------------------------
00632 void Triangulation::optimizeDelaunay() {
00633 
00634   // This function is also present in ttl where it is implemented
00635   // generically.
00636   // The implementation below is tailored for the half-edge data structure,
00637   // and is thus more efficient
00638 
00639   // Collect all interior edges (one half edge for each arc)
00640   bool skip_boundary_edges = true;
00641   list<Edge*>* elist = getEdges(skip_boundary_edges);
00642   
00643   // Assumes that elist has only one half-edge for each arc.
00644   bool cycling_check = true;
00645   bool optimal = false;
00646   list<Edge*>::const_iterator it;
00647   while(!optimal) {
00648     optimal = true;
00649     for (it = elist->begin(); it != elist->end(); ++it) {
00650       Edge* edge = *it;
00651       
00652       Dart dart(edge);
00653       // Constrained edges should not be swapped
00654       if (!edge->isConstrained() && ttl::swapTestDelaunay<TTLtraits>(dart, cycling_check)) {
00655         optimal = false;
00656         swapEdge(*edge);
00657       }
00658     }
00659   }
00660   delete elist;
00661 }

00666   
00667   const list<Edge*>& leadingEdges = getLeadingEdges();
00668   list<Edge*>::const_iterator it;
00669   for (it = leadingEdges.begin(); it != leadingEdges.end(); ++it) {
00670     Edge* edge = *it;
00671     
00672     // multiple checks, but only until found
00673     for (int i = 0; i < 3; ++i) {
00674       if (edge->getTwinEdge() != NULL) {
00675         
00676         if (!ttl::isBoundaryNode(Dart(edge)))
